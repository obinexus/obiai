// Four specific pathways through which bias enters ML models.

// Each is formalized as a detectable condition.

DEFINE BIAS_INFILTRATION_VECTORS AS:

VECTOR 1: data_collection_bias

definition = "Over/under-representation of population subgroups in dataset D"

detection:

PROCEDURE detect(D, protected_attribute) -> BIAS_FLAG:

group_representation = compute_group_frequencies(D, protected_attribute)

population_reference = load_population_distribution(domain)

deviation = KL_divergence(group_representation, population_reference)

Source Segment 9

IF deviation > KL_THRESHOLD:

RETURN BIAS_FLAG("Data collection bias — distributional mismatch")

VECTOR 2: feature_selection_bias

definition = "Choosing variables that correlate with protected attributes"

detection:

PROCEDURE detect(features, protected_attribute, D) -> BIAS_FLAG:

FOR EACH feature IN features:

correlation = compute_correlation(D[feature], D[protected_attribute])

IF abs(correlation) > CORRELATION_THRESHOLD:

RETURN BIAS_FLAG("Feature " + feature + " correlates with protected attribute")

VECTOR 3: label_bias

definition = "Historical diagnostic disparities encoded in ground truth labels"

Source Segment 10

// This is the most dangerous vector — bias embedded in the labels themselves

detection:

PROCEDURE detect(D, label_column, protected_attribute) -> BIAS_FLAG:

FOR EACH group IN unique(D[protected_attribute]):

group_data = filter(D, protected_attribute == group)

positive_rate = mean(group_data[label_column])

// If positive rate differs significantly across groups

// AND no clinical basis exists for the difference → label bias

IF abs(positive_rate - overall_positive_rate) > LABEL_DISPARITY_THRESHOLD:

RETURN BIAS_FLAG("Label bias — historical disparity in group " + group)

VECTOR 4: model_specification_bias

definition = "Algorithmic choices that amplify distributional imbalances"

Source Segment 11

examples:

- "Loss function indifferent to group performance"

- "Regularization penalizing minority group patterns"

- "Threshold selection optimized for majority group"

detection:

PROCEDURE detect(model, D, protected_attribute) -> BIAS_FLAG:

// Test if model amplifies input distributional imbalance

input_imbalance = measure_distributional_imbalance(D, protected_attribute)

output_imbalance = measure_distributional_imbalance(model.predictions, protected_attribute)

IF output_imbalance > input_imbalance * AMPLIFICATION_FACTOR:

RETURN BIAS_FLAG("Model specification amplifies distributional imbalance")

// COMPONENT 2: Structural Causal Modeling — Joint Probability Factorization

// Using the Bayesian network factorization theorem:

// P(X1, X2, ..., Xn) = Π_{i=1}^{n} P(Xi | Pa(Xi))

// Where Pa(Xi) = parent nodes of Xi in the DAG

PROCEDURE factorize_joint_distribution(dag, variables) -> FACTORED_DISTRIBUTION:

// Theorem: joint = product of each variable's conditional given its parents

Source Segment 14

joint = 1.0

FOR EACH variable X_i IN variables:

parents = dag.get_parents(X_i)

conditional = P(X_i GIVEN parents)

joint *= conditional

RETURN joint

// Factorization enables:

// 1. Identification of backdoor paths (bias paths through the DAG)

// 2. Explicit conditional independence assumptions

// 3. Targeted interventions on specific conditionals

PROCEDURE identify_backdoor_paths(dag, treatment_node, outcome_node, protected_attribute) -> LIST:

// A backdoor path from treatment to outcome via protected attribute = bias

Source Segment 15

paths = find_all_paths(dag, protected_attribute, outcome_node)

backdoor = filter(paths, lambda p: not p.blocked_by_conditioning)

RETURN backdoor

DEFINE HIERARCHICAL_BAYESIAN_PARAMETERS AS:

θ_parameters:

role = "True risk relationships — what we want to estimate"

// θ represents the causal relationships that exist INDEPENDENT of bias

// e.g., the true P(C|S) relationship uncontaminated by demographic confounds

φ_bias_factors:

role = "Explicit model of dataset skews"

Source Segment 16

// φ quantifies HOW MUCH and IN WHICH DIRECTION the dataset is biased

// By making φ explicit, we can integrate it out rather than letting it

// contaminate θ

marginalization:

// The hierarchical marginalization (previously Equations 3-5):

P_marginal(θ GIVEN D) = INTEGRAL over φ OF P(θ, φ GIVEN D) dφ

// Result: θ estimated FREE of the influence of φ

// Key: This only works when φ is EXPLICITLY modeled — not ignored

PROCEDURE hierarchical_bayesian_estimation(D, priors_θ, priors_φ) -> θ_unbiased:

// Step 1: Define the hierarchical model

// φ ~ P(φ) -- prior on bias factor

// θ | φ ~ P(θ|φ) -- conditional prior on parameters given bias

Source Segment 17

// D | θ, φ ~ P(D|θ, φ) -- likelihood: data given params and bias

// Step 2: Compute joint posterior

// P(θ, φ | D) ∝ P(D|θ,φ) · P(θ|φ) · P(φ)

// Step 3: Sample from joint posterior via MCMC or variational inference

posterior_samples = sample_joint_posterior(D, priors_θ, priors_φ)

// Step 4: Marginalize — integrate out φ from samples

θ_samples = [sample.θ FOR sample IN posterior_samples]

// These θ samples are now marginalized over φ — bias has been integrated out

// Step 5: Return posterior distribution over θ (not just point estimate)

RETURN posterior_distribution(θ_samples)

// Full distribution = explicit uncertainty quantification (not just MAP)
