\paragraph{Hamiltonian:}
$\hat{H} = \hat{T} + \hat{V} + \hat{C}$ where:
\begin{align}
\hat{T} &= \int d^3x \: \frac{1}{2m}|\nabla\Psi(\mathbf{x},t)|^2 \quad \text{(kinetic)} \\
\hat{V} &= \int d^3x \: V(\mathbf{x})\Psi^{\dagger}(\mathbf{x},t)\Psi(\mathbf{x},t) \quad \text{(potential)} \\
\hat{C} &= \int d^3x d^3y \: J(\mathbf{x},\mathbf{y})\Psi^{\dagger}(\mathbf{x},t)\Psi(\mathbf{y},t) \quad \text{(coherence)}
\end{align}
