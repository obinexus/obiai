// ── ASSUMPTION 1: ENVIRONMENT STATIONARITY ───────────────────

// ASSUMPTION 1 (Environment Stationarity): // The environment distribution E exhibits WEAK STATIONARITY: // // E[Xₜ] = μ (constant mean) // Cov(Xₜ, Xₜ₊ₖ) = γ(k) for all t (covariance depends only on lag k) // // Interpretation: the statistical properties of the environment // do not drift over time — a necessary condition for convergence // guarantees in AEGIS-PROOF-3.2.

DEFINE EnvironmentStationarity AS: // Weak stationarity conditions on environment distribution E

PROCEDURE VerifyMeanStationarity(E: EnvironmentDistribution,
                                  t_samples: List[INT],
                                  tolerance: REAL) -> BOOL:
    // Check E[Xₜ] ≈ μ (constant) for all sampled time points t
    means := []
    FOR EACH t IN t_samples:
        means.APPEND(E.sample_mean(at_time=t))
    END FOR
    mean_variance := VARIANCE(means)
    IF mean_variance > tolerance:
        EMIT WARNING "Assumption 1 violated: mean drift detected | var=" + mean_variance
        RETURN FALSE
    END IF
    RETURN TRUE
END PROCEDURE

PROCEDURE VerifyCovarianceStationarity(E: EnvironmentDistribution,
                                        lag_k: INT, t1: INT, t2: INT,
                                        tolerance: REAL) -> BOOL:
    // Check Cov(Xₜ₁, Xₜ₁₊ₖ) ≈ Cov(Xₜ₂, Xₜ₂₊ₖ) = γ(k) for all t
    cov_t1 := E.covariance(t1, t1 + lag_k)
    cov_t2 := E.covariance(t2, t2 + lag_k)
    IF ABS(cov_t1 - cov_t2) > tolerance:
        EMIT WARNING "Assumption 1 violated: covariance not time-invariant at lag=" + lag_k
        RETURN FALSE
    END IF
    RETURN TRUE
END PROCEDURE

END DEFINE

// ── ASSUMPTION 2: COST FUNCTION PROPERTIES ───────────────────

// ASSUMPTION 2 (Cost Function Properties): // Runtime and error cost functions must satisfy: // // (1) LIPSCHITZ: |C_runtime(m₁) − C_runtime(m₂)| ≤ Lᵣ · ‖m₁ − m₂‖ // (2) MONOTONE: ∂C_error(m) / ∂p_conf ≤ 0 ∀m ∈ {Flash, Filter} // (3) BOUNDED: 0 ≤ C_runtime(m), C_error(m) ≤ M < ∞

DEFINE CostFunctionProperties AS: Lr : REAL // Lipschitz constant for runtime cost M : REAL // upper bound on all cost functions

PROCEDURE VerifyLipschitz(C_runtime: Function, m1: Mode, m2: Mode,
                           Lr: REAL) -> BOOL:
    // |C_runtime(m₁) − C_runtime(m₂)| ≤ Lᵣ · ‖m₁ − m₂‖
    lhs := ABS(C_runtime(m1) - C_runtime(m2))
    rhs := Lr * NormDifference(m1, m2)
    IF lhs > rhs:
        EMIT WARNING "Assumption 2(Lipschitz) violated: " + lhs + " > " + rhs
        RETURN FALSE
    END IF
    RETURN TRUE
END PROCEDURE

PROCEDURE VerifyMonotone(C_error: Function, m: Mode,
                          p_conf_range: List[REAL]) -> BOOL:
    // ∂C_error(m) / ∂p_conf ≤ 0 — error cost DECREASES as confidence increases
    // Numerical check: cost values decrease with increasing p_conf
    prev_cost := POSITIVE_INFINITY
    FOR EACH p IN SORTED(p_conf_range, order=ASCENDING):
        cost := C_error(m, p)
        IF cost > prev_cost + 1e-10:
            EMIT WARNING "Assumption 2(Monotone) violated at p=" + p +
                         " for mode " + m
            RETURN FALSE
        END IF
        prev_cost := cost
    END FOR
    RETURN TRUE
END PROCEDURE

PROCEDURE VerifyBounded(C_runtime: Function, C_error: Function,
                         M: REAL, test_modes: List[Mode]) -> BOOL:
    // 0 ≤ C_runtime(m), C_error(m) ≤ M for all modes
    FOR EACH m IN test_modes:
        c_run := C_runtime(m)
        c_err := C_error(m)
        IF c_run < 0 OR c_run > M:
            EMIT WARNING "Assumption 2(Bounded) violated: C_runtime(" + m + ")=" + c_run
            RETURN FALSE
        END IF
        IF c_err < 0 OR c_err > M:
            EMIT WARNING "Assumption 2(Bounded) violated: C_error(" + m + ")=" + c_err
            RETURN FALSE
        END IF
    END FOR
    RETURN TRUE
END PROCEDURE

END DEFINE

PROCEDURE VerifyContinuity(DAG_cost: Function, v: Verb, n: Noun,
                            delta: REAL) -> BOOL:
    // Numerical continuity check: small perturbation → small cost change
    cost_base   := DAG_cost(v, n)
    cost_perturb := DAG_cost(Perturb(v, delta), Perturb(n, delta))
    IF ABS(cost_perturb - cost_base) > 10 * delta:

        RETURN FALSE
    END IF
    RETURN TRUE
END PROCEDURE

PROCEDURE VerifyGradientBound(DAG_cost: Function, v: Verb, n: Noun,

PROCEDURE VerifyConvexity(DAG_cost: Function, v: Verb, n: Noun) -> BOOL:
    // ∇² DAG_cost ⪰ 0 (all eigenvalues of Hessian ≥ 0)
    hessian     := ComputeHessian(DAG_cost, v, n)

            RETURN FALSE
        END IF
    END FOR
    RETURN TRUE
END PROCEDURE

// ============================================================ // FILE: aegis_proof_3_1_3_2_M2_filterflash_monotonicity.psc.txt // MODULE 2 OF 5 — AEGIS-PROOF-3.1: Filter-Flash Monotonicity // SOURCE: "AEGIS-PROOF-3.1 & 3.2: Mathematical Verification Suite" // AUTHOR: Nnamdi Michael Okpala // ORG: OBINexus Computing — AEGIS Project Team // DATE: August 2025 // ============================================================

// ── MODE DEFINITIONS ─────────────────────────────────────────

DEFINE CognitionMode AS ENUM: FILTER // persistent symbolic reasoning — lower error at high confidence FLASH // ephemeral working memory — higher error at low confidence END DEFINE

// Epistemic confidence variable: // p = p_conf ∈ [0, 1] // p = 0.954 → 95.4% threshold (optimal selection boundary per Corollary 1)

CONSTANT P_CONF_OPTIMAL := 0.954 // AEGIS epistemic confidence threshold

// ── THEOREM 1: FILTER-FLASH MONOTONICITY ─────────────────────

// THEOREM 1 (Filter-Flash Monotonicity): // // Under Assumptions 1–3, for fixed environment distribution // and monotone cost functions, increasing epistemic confidence // p_conf MONOTONICALLY INCREASES the advantage of Filter over Flash. // // Formally: Δ(p) = E[C_Flash] − E[C_Filter] is non-decreasing in p. // i.e., dΔ(p)/dp ≥ 0

// ── STEP 1: COST ADVANTAGE FUNCTION DECOMPOSITION ────────────

// Δ(p) = E[C_Flash(p)] − E[C_Filter(p)] // // Expanding via total cost formulation: // Δ(p) = (E[C_Flash_runtime] − E[C_Filter_runtime]) ← constant term // + (E[C_Flash_error(p)] − E[C_Filter_error(p)]) ← Δ_error(p)

PROCEDURE DecomposeCostAdvantage(p_conf: REAL, C_runtime: Function, C_error: Function, dag: EpistemicDAG) -> CostAdvantageDecomposition:

// Term 1: Runtime cost differential (CONSTANT — independent of p_conf)
runtime_flash  := EXPECTATION(C_runtime(FLASH))
runtime_filter := EXPECTATION(C_runtime(FILTER))
constant_term  := runtime_flash - runtime_filter

// Term 2: Error cost differential (p_conf-DEPENDENT)
error_flash  := EXPECTATION(C_error_total(FLASH,  p_conf, dag))
error_filter := EXPECTATION(C_error_total(FILTER, p_conf, dag))
delta_error  := error_flash - error_filter

// Total advantage
delta := constant_term + delta_error

decomp := CostAdvantageDecomposition(
    constant_term  = constant_term,
    delta_error    = delta_error,
    total_delta    = delta,
    p_conf         = p_conf
)
RETURN decomp

END PROCEDURE

// ── STEP 2: ERROR COST DIFFERENTIAL ANALYSIS ─────────────────

// Error cost for mode m, weighted by DAG cost: // C_error^m(p) = Σ_{v,n} DAG_cost(v, n) · P(error | v, n, m, p) // // Key property (from Assumption 2 monotone condition): // ∂P(error | v, n, Filter, p) / ∂p ≤ ∂P(error | v, n, Flash, p) / ∂p // // Rationale: // Filter uses persistent symbolic reasoning → error probability // decreases FASTER with confidence than Flash's ephemeral working memory.

PROCEDURE C_error_total(mode: CognitionMode, p_conf: REAL, dag: EpistemicDAG) -> REAL: // C_error^m(p) = Σ_{v,n} DAG_cost(v, n) · P(error | v, n, m, p) total := 0.0 FOR EACH (v, n) IN dag.verb_noun_pairs: dag_cost_vn := dag.cost(v, n) p_error_vn := P_error(v, n, mode, p_conf) total := total + dag_cost_vn * p_error_vn END FOR RETURN total END PROCEDURE

PROCEDURE P_error(v: Verb, n: Noun, mode: CognitionMode, p_conf: REAL) -> REAL: // Error probability for mode m at confidence p. // Flash: higher baseline error, slower improvement with p_conf // Filter: lower baseline error, faster improvement with p_conf

MATCH mode:
    CASE FLASH:
        // Ephemeral working memory — error probability higher at low confidence,
        // decreasing more slowly: ∂P_error/∂p steeper (larger magnitude) than Filter
        return BaseErrorRate(v, n, FLASH) * (1.0 - p_conf)

    CASE FILTER:
        // Persistent symbolic reasoning — error probability lower overall,
        // decreasing faster with confidence gains
        return BaseErrorRate(v, n, FILTER) * (1.0 - p_conf)^2

END MATCH
// NOTE: The exact functional form is implementation-specific.
// The critical property (Assumption 2) is that ∂P_error/∂p is MORE
// negative for Filter than Flash — Filter benefits more from confidence.

END PROCEDURE

PROCEDURE VerifyErrorDerivativeOrdering(v: Verb, n: Noun, p_conf: REAL) -> BOOL: // Verify: ∂P(error|v,n,Filter,p)/∂p ≤ ∂P(error|v,n,Flash,p)/∂p // i.e., Filter's error decreases at LEAST as fast as Flash's

dp_filter := NumericalDerivative(P_error, args=(v,n,FILTER), wrt=p_conf)
dp_flash  := NumericalDerivative(P_error, args=(v,n,FLASH),  wrt=p_conf)

// Both should be ≤ 0 (error decreases with confidence)
// Filter's derivative should be MORE negative (or equal)
IF dp_filter > dp_flash + 1e-10:
    EMIT WARNING "Derivative ordering violated at (v=" + v + ", n=" + n +
                 ", p=" + p_conf + ")"
    RETURN FALSE
END IF
RETURN TRUE

END PROCEDURE

// ── STEP 3: MONOTONICITY OF ERROR DIFFERENTIAL ───────────────

// d(Δ_error(p)) / dp = Σ_{v,n} DAG_cost(v,n) · d/dp[P(error|v,n,Flash,p) // − P(error|v,n,Filter,p)] // // Since DAG_cost(v,n) ≥ 0 (by construction) AND // d/dp[P_Flash_error − P_Filter_error] ≥ 0 (from derivative ordering): // // d(Δ_error(p)) / dp ≥ 0 ✓

PROCEDURE VerifyErrorDifferentialMonotonicity(dag: EpistemicDAG, p_conf: REAL) -> BOOL: // Verify that d(Δ_error)/dp ≥ 0 at p_conf

derivative_sum := 0.0

FOR EACH (v, n) IN dag.verb_noun_pairs:
    dag_cost_vn := dag.cost(v, n)
    ASSERT dag_cost_vn >= 0.0    // DAG cost non-negative by construction

    d_flash  := NumericalDerivative(P_error, args=(v,n,FLASH),  wrt=p_conf)
    d_filter := NumericalDerivative(P_error, args=(v,n,FILTER), wrt=p_conf)
    d_diff   := d_flash - d_filter    // d/dp[P_Flash − P_Filter]
    ASSERT d_diff >= -1e-10           // ≥ 0 from derivative ordering

    derivative_sum := derivative_sum + dag_cost_vn * d_diff
END FOR

IF derivative_sum < -1e-10:
    EMIT ERROR "d(Δ_error)/dp < 0 — monotonicity of error differential violated"
    RETURN FALSE
END IF
RETURN TRUE

END PROCEDURE

// ── STEP 4: COMPLETE MONOTONICITY PROOF ──────────────────────

// dΔ(p)/dp = d(constant_term)/dp + d(Δ_error(p))/dp // = 0 + d(Δ_error(p))/dp // ≥ 0 // // Therefore Δ(p) is non-decreasing in p. □

PROCEDURE ProveFilterFlashMonotonicity(dag: EpistemicDAG, C_runtime: Function, C_error: Function, p_range: List[REAL]) -> MonotonicityProof:

proof := NEW MonotonicityProof(theorem="AEGIS-PROOF-3.1")

// Verify derivative ordering holds at each test p value
proof.derivative_ordering_ok := TRUE
FOR EACH p IN p_range:
    FOR EACH (v, n) IN dag.verb_noun_pairs:
        ok := VerifyErrorDerivativeOrdering(v, n, p)
        IF NOT ok:
            proof.derivative_ordering_ok := FALSE
        END IF
    END FOR
END FOR

// Verify error differential monotonicity at each test p
proof.error_diff_monotone := TRUE
FOR EACH p IN p_range:
    ok := VerifyErrorDifferentialMonotonicity(dag, p)
    IF NOT ok:
        proof.error_diff_monotone := FALSE
    END IF
END FOR

// Verify Δ(p) is non-decreasing: Δ(p₁) ≤ Δ(p₂) for p₁ < p₂
proof.delta_nondecreasing := TRUE
prev_delta := -POSITIVE_INFINITY
FOR EACH p IN SORTED(p_range, ASCENDING):
    decomp := DecomposeCostAdvantage(p, C_runtime, C_error, dag)
    IF decomp.total_delta < prev_delta - 1e-10:
        proof.delta_nondecreasing := FALSE
        EMIT WARNING "Δ(p) decreased at p=" + p
    END IF
    prev_delta := decomp.total_delta
END FOR

proof.theorem_holds := proof.derivative_ordering_ok AND
                       proof.error_diff_monotone AND
                       proof.delta_nondecreasing

IF proof.theorem_holds:
    EMIT INFO "AEGIS-PROOF-3.1 VERIFIED: Filter-Flash Monotonicity confirmed"
ELSE:
    EMIT ERROR "AEGIS-PROOF-3.1 FAILED: Monotonicity not established"
END IF

RETURN proof

END PROCEDURE

PROCEDURE SelectOptimalMode(p_conf: REAL) -> CognitionMode: // Mode selection rule derived from Theorem 1: // Because Δ(p) is non-decreasing, once p_conf crosses P_CONF_OPTIMAL, // Filter mode is guaranteed to have lower expected cost than Flash.

// ============================================================ // FILE: aegis_proof_3_1_3_2_M3_hybrid_convergence.psc.txt // MODULE 3 OF 5 — AEGIS-PROOF-3.2: Hybrid Mode Convergence // SOURCE: "AEGIS-PROOF-3.1 & 3.2: Mathematical Verification Suite" // AUTHOR: Nnamdi Michael Okpala // ORG: OBINexus Computing — AEGIS Project Team // DATE: August 2025 // ============================================================

// ── THEOREM 2: HYBRID MODE CONVERGENCE ───────────────────────

// THEOREM 2 (Hybrid Mode Convergence): // // Under bounded update steps and diminishing learning rate, repeated // hybrid-mode updates converge to a fixed point minimizing: // // J(v, n) = DAG_cost(v,n) + λ·C_G(v,n) + μ·TC(v,n) // // Formally: let {(vₙ, nₙ)} be the sequence generated by hybrid updates. // Then: // lim_{n→∞} DAG_cost(vₙ, nₙ) + λ·C_G(vₙ, nₙ) + μ·TC(vₙ, nₙ) = J // // Proof method: Robbins-Monro stochastic approximation + // Lyapunov function stability analysis.

// ── UPDATE RULE PARAMETER CONSTANTS ─────────────────────────

CONSTANT ALPHA_0 := 0.01 // initial learning rate α₀ CONSTANT BETA_DECAY := 0.6 // decay exponent β ∈ (0.5, 1] CONSTANT NOISE_VARIANCE := 1.0 // σ² for Gaussian noise ξₙ CONSTANT EPSILON_TOL := 1e-6 // convergence tolerance on ‖∇J‖

// Objective function components (user-supplied regularization weights): // λ — weight for grammar cost C_G // μ — weight for transition cost TC

// ── STEP 1: HYBRID UPDATE FORMULATION ────────────────────────

// Update rule (stochastic gradient descent with noise): // (vₙ₊₁, nₙ₊₁) = (vₙ, nₙ) − αₙ · ∇J(vₙ, nₙ) + ξₙ // // where: // J(v, n) = DAG_cost(v, n) + λ·C_G(v, n) + μ·TC(v, n) // αₙ = α₀ / nᵝ (diminishing learning rate, β ∈ (0.5, 1]) // ξₙ ~ N(0, σ²I) (bounded Gaussian noise)

DEFINE ObjectiveFunction J AS: PROCEDURE Evaluate(v: Verb, n: Noun, lambda_w: REAL, mu_w: REAL, dag: EpistemicDAG) -> REAL: // J(v, n) = DAG_cost(v,n) + λ·C_G(v,n) + μ·TC(v,n) dag_term := dag.cost(v, n) cg_term := GrammarCost(v, n) // C_G: grammatical regularity cost tc_term := TransitionCost(v, n) // TC: transition cost

    RETURN dag_term + (lambda_w * cg_term) + (mu_w * tc_term)
END PROCEDURE

PROCEDURE Gradient(v: Verb, n: Noun,
                   lambda_w: REAL, mu_w: REAL,
                   dag: EpistemicDAG) -> GradientVector:
    // ∇J(v, n) = ∇DAG_cost + λ·∇C_G + μ·∇TC
    grad_dag := NumericalGradient(dag.cost, v, n)
    grad_cg  := NumericalGradient(GrammarCost, v, n)
    grad_tc  := NumericalGradient(TransitionCost, v, n)

    RETURN grad_dag + (lambda_w * grad_cg) + (mu_w * grad_tc)
END PROCEDURE

END DEFINE

PROCEDURE DiminishingLearningRate(n: INT) -> REAL: // αₙ = α₀ / nᵝ (β > 0.5 ensures Σα²ₙ < ∞) IF n == 0: RETURN ALPHA_0 END IF RETURN ALPHA_0 / (n ^ BETA_DECAY) END PROCEDURE

PROCEDURE SampleNoise(dim: INT) -> Vector: // ξₙ ~ N(0, σ²I) — bounded Gaussian noise RETURN GAUSSIAN_SAMPLE(mean=ZERO_VECTOR(dim), variance=NOISE_VARIANCE) END PROCEDURE

PROCEDURE HybridUpdateStep(v_n: Verb, n_n: Noun, iteration: INT, lambda_w: REAL, mu_w: REAL, dag: EpistemicDAG) -> (Verb, Noun): // (vₙ₊₁, nₙ₊₁) = (vₙ, nₙ) − αₙ·∇J(vₙ, nₙ) + ξₙ

alpha_n  := DiminishingLearningRate(iteration)
grad_J   := J.Gradient(v_n, n_n, lambda_w, mu_w, dag)
noise    := SampleNoise(dim=2)     // 2D: verb + noun dimensions

// Gradient step
v_next := v_n - alpha_n * grad_J.v_component + noise[0]
n_next := n_n - alpha_n * grad_J.n_component + noise[1]

RETURN (v_next, n_next)

END PROCEDURE

// ── STEP 2: ROBBINS-MONRO CONDITIONS ─────────────────────────

// Standard conditions for Robbins-Monro convergence: // (a) Σ αₙ = ∞ (learning rates sum to infinity) // (b) Σ α²ₙ < ∞ (squared learning rates are summable) // (c) ‖∇J(v,n)‖ ≤ G (bounded gradients — from Assumption 3) // (d) J(v,n) is convex (from Assumption 3)

PROCEDURE VerifyRobbinsMonroConditions(G: REAL) -> RobbinsMonroReport: report := NEW RobbinsMonroReport()

// (a) Σ αₙ = ∞ : verified analytically for αₙ = α₀/nᵝ with β ≤ 1
//     Σ 1/nᵝ diverges for β ≤ 1 (p-series, p=β≤1)
report.summable_rates := TRUE    // β=0.6 ≤ 1 → diverges ✓
EMIT INFO "RM(a): Σαₙ = ∞ — confirmed (β=" + BETA_DECAY + " ≤ 1)"

// (b) Σ α²ₙ < ∞ : verified analytically for β > 0.5
//     Σ 1/n^(2β) converges for 2β > 1, i.e., β > 0.5
report.square_summable := (BETA_DECAY > 0.5)
IF report.square_summable:
    EMIT INFO "RM(b): Σα²ₙ < ∞ — confirmed (β=" + BETA_DECAY + " > 0.5)"
ELSE:
    EMIT ERROR "RM(b): β ≤ 0.5 — square summability VIOLATED"
END IF

// (c) Bounded gradients — from Assumption 3 (verified in Module 1)
report.bounded_gradients := (G < POSITIVE_INFINITY)
EMIT INFO "RM(c): ‖∇J‖ ≤ G=" + G + " — from Assumption 3"

// (d) Convexity — from Assumption 3 (verified in Module 1)
report.convexity := TRUE    // carried forward from Assumption 3 verification
EMIT INFO "RM(d): J(v,n) convex — from Assumption 3"

report.all_conditions_met := report.summable_rates AND
                              report.square_summable AND
                              report.bounded_gradients AND
                              report.convexity
RETURN report

END PROCEDURE

// ── STEP 3: LYAPUNOV FUNCTION CONSTRUCTION ───────────────────

// Define the Lyapunov function: // Vₙ = J(vₙ, nₙ) − J* // // Vₙ measures the "distance" from current objective value to global minimum. // For convergence: we need E[Vₙ] → 0.

PROCEDURE LyapunovFunction(v_n: Verb, n_n: Noun, J_star: REAL, lambda_w: REAL, mu_w: REAL, dag: EpistemicDAG) -> REAL: // Vₙ = J(vₙ, nₙ) − J J_n := J.Evaluate(v_n, n_n, lambda_w, mu_w, dag) V_n := J_n - J_star ASSERT V_n >= 0.0 // J is global minimum → J_n ≥ J* RETURN V_n END PROCEDURE

// ── STEP 4: TAYLOR EXPANSION AND CONVERGENCE BOUND ───────────

// Using second-order Taylor expansion around (vₙ, nₙ): // // E[Vₙ₊₁] ≤ Vₙ − αₙ·‖∇J(vₙ,nₙ)‖² + αₙ²·L²·‖∇J(vₙ,nₙ)‖² + αₙ²·σ²·C // // For sufficiently large n, since αₙ = O(n⁻ᵝ) with β > 0.5: // // E[Vₙ₊₁] ≤ Vₙ − αₙ·‖∇J‖²·(1 − αₙ·L²) + αₙ²·σ²·C

PROCEDURE ComputeExpectedLyapunovBound(V_n: REAL, alpha_n: REAL, grad_norm_sq: REAL, L: REAL, sigma_sq: REAL, C_noise: REAL) -> REAL: // E[Vₙ₊₁] ≤ Vₙ − αₙ·‖∇J‖² + αₙ²·L²·‖∇J‖² + αₙ²·σ²·C // // Compact form: ≤ Vₙ − αₙ·‖∇J‖²·(1 − αₙ·L²) + αₙ²·σ²·C

descent_term  := alpha_n * grad_norm_sq
curvature_cor := (alpha_n ^ 2) * (L ^ 2) * grad_norm_sq
noise_term    := (alpha_n ^ 2) * sigma_sq * C_noise

bound := V_n - descent_term + curvature_cor + noise_term

// For the bound to guarantee descent: (1 − αₙ·L²) > 0
//   → αₙ < 1/L²  (satisfied for large n since αₙ → 0)
IF alpha_n * (L ^ 2) >= 1.0:
    EMIT WARNING "Taylor bound: αₙ·L² ≥ 1 at this step — descent not guaranteed yet"
END IF

RETURN bound

END PROCEDURE

// ── STEP 5: ALMOST SURE CONVERGENCE ──────────────────────────

// Since Σα²ₙ < ∞ and noise terms vanish asymptotically: // // Σ_{n=1}^∞ αₙ·‖∇J(vₙ,nₙ)‖² < ∞ a.s. // // This implies ‖∇J(vₙ,nₙ)‖ → 0 a.s. // By convexity: ‖∇J‖ → 0 implies J(vₙ,nₙ) → J* a.s.

PROCEDURE RunHybridConvergenceLoop(v0: Verb, n0: Noun, J_star: REAL, lambda_w: REAL, mu_w: REAL, dag: EpistemicDAG, max_iterations: INT) -> ConvergenceResult:

v_n := v0
n_n := n0
grad_sq_sum := 0.0    // accumulator for Σ αₙ·‖∇J‖²

history := [(v_n, n_n, J.Evaluate(v_n, n_n, lambda_w, mu_w, dag))]

FOR iteration := 1 TO max_iterations:
    alpha_n   := DiminishingLearningRate(iteration)
    grad_J    := J.Gradient(v_n, n_n, lambda_w, mu_w, dag)
    grad_norm := NORM(grad_J)

    // Accumulate Σ αₙ·‖∇J‖² (must be finite a.s. for convergence)
    grad_sq_sum := grad_sq_sum + alpha_n * (grad_norm ^ 2)

    // Apply hybrid update
    (v_n, n_n) := HybridUpdateStep(v_n, n_n, iteration,
                                    lambda_w, mu_w, dag)

    J_current := J.Evaluate(v_n, n_n, lambda_w, mu_w, dag)
    history.APPEND((v_n, n_n, J_current))

    EMIT LOG "Iter " + iteration + " | αₙ=" + alpha_n +
             " | ‖∇J‖=" + grad_norm + " | J=" + J_current

    // Convergence check: ‖∇J‖ ≤ ε_tol
    IF grad_norm <= EPSILON_TOL:
        EMIT INFO "CONVERGENCE at iteration " + iteration +
                  " | J=" + J_current + " ≈ J*=" + J_star
        RETURN ConvergenceResult(converged=TRUE, iterations=iteration,
                                 final_v=v_n, final_n=n_n,
                                 final_J=J_current, J_star=J_star,
                                 grad_sq_sum=grad_sq_sum, history=history)
    END IF
END FOR

EMIT WARNING "MAX_ITERATIONS reached without convergence"
RETURN ConvergenceResult(converged=FALSE, iterations=max_iterations,
                         final_v=v_n, final_n=n_n,
                         final_J=J.Evaluate(v_n, n_n, lambda_w, mu_w, dag),
                         J_star=J_star, history=history)

END PROCEDURE

// ── COROLLARY 2: CONVERGENCE RATE ────────────────────────────

// COROLLARY 2 (Convergence Rate): // Under ADDITIONAL STRONG CONVEXITY, convergence rate is O(1/n). // // Strong convexity condition: // ∇²J(v,n) ⪰ μ·I for some μ > 0 (all eigenvalues ≥ μ)

PROCEDURE VerifyStrongConvexity(J_fn: Function, v: Verb, n: Noun, mu_strong: REAL) -> BOOL: // ∇²J ⪰ μ·I ↔ all eigenvalues of Hessian ≥ μ_strong hessian := ComputeHessian(J_fn, v, n) eigenvalues := ComputeEigenvalues(hessian) FOR EACH λ IN eigenvalues: IF λ < mu_strong - 1e-10: EMIT WARNING "Strong convexity violated: λ=" + λ + " < μ=" + mu_strong RETURN FALSE END IF END FOR EMIT INFO "Strong convexity verified: all eigenvalues ≥ μ=" + mu_strong RETURN TRUE END PROCEDURE

PROCEDURE EstimateConvergenceRate(history: List[(Verb, Noun, REAL)], J_star: REAL) -> REAL: // Measure empirical convergence rate from history. // Under strong convexity, |J(n) − J| = O(1/n). // Fit log|J(n) − J| vs log(n) to estimate exponent.

n_last := LENGTH(history) - 1
IF n_last < 10:
    EMIT WARNING "Too few iterations for rate estimation"
    RETURN NULL
END IF

// Log-log regression: log|J(n)-J*| ≈ -rate * log(n) + const
log_errors := []
log_iters  := []
FOR idx := 1 TO n_last:
    (v, n, J_val) := history[idx]
    err := ABS(J_val - J_star)
    IF err > 0:
        log_errors.APPEND(LOG(err))
        log_iters.APPEND(LOG(idx))
    END IF
END FOR

slope := LinearRegressionSlope(log_iters, log_errors)
rate  := -slope    // expect rate ≈ 1.0 for O(1/n)
EMIT INFO "Empirical convergence rate: O(1/n^" + rate + ")"
RETURN rate

END PROCEDURE

// ------------------------------------------------------------ // ALGORITHM 1: AEGIS-VERIFIED HYBRID MODE IMPLEMENTATION // ------------------------------------------------------------

// This is the primary production algorithm integrating: // - Epistemic confidence computation (AEGIS-PROOF-3.1) // - Monotonicity verification at each step // - Diminishing learning rate schedule (AEGIS-PROOF-3.2) // - Gradient descent update // - DIRAM constraint enforcement (ε(transition) ≤ 0.6) // // INPUTS: Initial state (v₀, n₀), confidence threshold θ = 0.954 // OUTPUTS: Converged optimal configuration (v, n)

PROCEDURE AEGISVerifiedHybridMode(v0: Verb, n0: Noun, dag: EpistemicDAG, lambda_w: REAL, mu_w: REAL, G: REAL, L: REAL) -> HybridModeResult:

// ── INITIALIZATION ─────────────────────────────────────
n_iter   := 0
v_n      := v0
n_n      := n0
alpha_0  := ALPHA_0                      // 0.01
beta_dec := BETA_DECAY                   // 0.6
theta    := P_CONF_OPTIMAL               // 0.954
epsilon_tol := EPSILON_TOL              // 1e-6

trace_log := []
J_history := []

EMIT INFO "=== AEGIS-VERIFIED HYBRID MODE STARTING ==="
EMIT INFO "Initial state: v₀=" + v0 + " n₀=" + n0
EMIT INFO "θ=" + theta + " | α₀=" + alpha_0 + " | β=" + beta_dec

// ── VERIFY ROBBINS-MONRO CONDITIONS BEFORE LOOP ─────────
rm_check := VerifyRobbinsMonroConditions(G)
IF NOT rm_check.all_conditions_met:
    EMIT ERROR "Robbins-Monro conditions not satisfied — aborting"
    RETURN HybridModeResult(success=FALSE, reason="RM conditions failed")
END IF

// ── MAIN CONVERGENCE LOOP ───────────────────────────────
grad_norm := POSITIVE_INFINITY    // initialize above threshold

WHILE grad_norm > epsilon_tol:

    n_iter := n_iter + 1

    // STEP A: Compute epistemic confidence at current state
    p_conf := ComputeConfidence(v_n, n_n, dag)

    // STEP B: Apply AEGIS-PROOF-3.1 — verify monotonicity condition
    //         (ensures mode selection is optimal at this confidence level)
    mode   := SelectOptimalMode(p_conf)     // FILTER if p≥0.954, else FLASH
    mono_ok := VerifyErrorDifferentialMonotonicity(dag, p_conf)
    IF NOT mono_ok:
        EMIT WARNING "Monotonicity condition failed at iter " + n_iter +
                     " — continuing under reduced confidence"
    END IF

    // STEP C: Update learning rate αₙ = α₀ / nᵝ
    alpha_n := DiminishingLearningRate(n_iter)

    // STEP D: Gradient step
    //   (vₙ₊₁, nₙ₊₁) = (vₙ, nₙ) − αₙ · ∇J(vₙ, nₙ)
    grad_J := J.Gradient(v_n, n_n, lambda_w, mu_w, dag)
    v_next := v_n - alpha_n * grad_J.v_component
    n_next := n_n - alpha_n * grad_J.n_component

    // STEP E: Add bounded noise ξₙ ~ N(0, σ²I)
    noise      := SampleNoise(dim=2)
    v_next     := v_next + noise[0]
    n_next     := n_next + noise[1]

    // STEP F: Verify DIRAM constraint ε(transition) ≤ 0.6
    diram_ok := VerifyDIRAMCompatibility(v_next, n_next,
                                          J_STAR_APPROX,
                                          lambda_w, mu_w, dag, n_iter)
    IF NOT diram_ok:
        // DIRAM violation: clamp step to respect ε bound
        v_next := LERP(v_n, v_next, t=0.5)    // halve the step
        n_next := LERP(n_n, n_next, t=0.5)
        EMIT INFO "DIRAM step clamped at iter " + n_iter
    END IF

    // STEP G: Commit update
    v_n := v_next
    n_n := n_next

    // STEP H: Compute gradient norm for convergence check
    grad_J_new := J.Gradient(v_n, n_n, lambda_w, mu_w, dag)
    grad_norm  := NORM(grad_J_new)

    // Record trace entry
    trace_record := PrintAndTraceVerification(v_n, n_n, p_conf,
                                               n_iter, dag, lambda_w, mu_w)
    trace_log.APPEND(trace_record)
    J_history.APPEND(J.Evaluate(v_n, n_n, lambda_w, mu_w, dag))

    EMIT LOG "n=" + n_iter + " | αₙ=" + alpha_n + " | ‖∇J‖=" + grad_norm +
             " | mode=" + mode + " | DIRAM=" + (IF diram_ok THEN "✓" ELSE "⚠")

END WHILE   // convergence achieved when ‖∇J‖ ≤ ε_tol

// ── RETURN CONVERGED CONFIGURATION ─────────────────────
J_star_final := J.Evaluate(v_n, n_n, lambda_w, mu_w, dag)
EMIT INFO "=== CONVERGENCE ACHIEVED | iter=" + n_iter +
          " | J*≈" + J_star_final + " ==="
EMIT INFO "Convergence guarantee: AEGIS-PROOF-3.2"

RETURN HybridModeResult(
    success    = TRUE,
    v_star     = v_n,
    n_star     = n_n,
    J_star     = J_star_final,
    iterations = n_iter,
    trace_log  = trace_log,
    J_history  = J_history
)

END PROCEDURE

PROCEDURE ComputeConfidence(v: Verb, n: Noun, dag: EpistemicDAG) -> REAL: // Compute epistemic confidence p_conf at (v, n) in the DAG. // Higher confidence → smoother posterior distribution → lower entropy. H := ComputeNodeEntropy(dag.node_at(v, n)) H_max := LOG2(dag.n_interpretations) // maximum entropy for n classes p_conf := 1.0 - (H / MAX(H_max, 1e-12)) // normalized: 1 = fully confident RETURN CLAMP(p_conf, 0.0, 1.0) END PROCEDURE
