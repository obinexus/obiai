// ------------------------------------------------------------

// This is the primary production algorithm integrating:
//   - Epistemic confidence computation (AEGIS-PROOF-3.1)
//   - Monotonicity verification at each step
//   - Diminishing learning rate schedule (AEGIS-PROOF-3.2)
//   - Gradient descent update
//   - DIRAM constraint enforcement (ε(transition) ≤ 0.6)
//
// INPUTS:  Initial state (v₀, n₀), confidence threshold θ = 0.954
// OUTPUTS: Converged optimal configuration (v*, n*)

PROCEDURE AEGISVerifiedHybridMode(v0: Verb, n0: Noun,
                                   dag: EpistemicDAG,
                                   lambda_w: REAL, mu_w: REAL,
                                   G: REAL, L: REAL) -> HybridModeResult:

    // ── INITIALIZATION ─────────────────────────────────────
    n_iter   := 0
    v_n      := v0
    n_n      := n0
    alpha_0  := ALPHA_0                      // 0.01
    beta_dec := BETA_DECAY                   // 0.6
    theta    := P_CONF_OPTIMAL               // 0.954
    epsilon_tol := EPSILON_TOL              // 1e-6

    trace_log := []
    J_history := []

    EMIT INFO "=== AEGIS-VERIFIED HYBRID MODE STARTING ==="
    EMIT INFO "Initial state: v₀=" + v0 + " n₀=" + n0
    EMIT INFO "θ=" + theta + " | α₀=" + alpha_0 + " | β=" + beta_dec

    // ── VERIFY ROBBINS-MONRO CONDITIONS BEFORE LOOP ─────────
    rm_check := VerifyRobbinsMonroConditions(G)
    IF NOT rm_check.all_conditions_met:
        EMIT ERROR "Robbins-Monro conditions not satisfied — aborting"
        RETURN HybridModeResult(success=FALSE, reason="RM conditions failed")
    END IF

    // ── MAIN CONVERGENCE LOOP ───────────────────────────────
    grad_norm := POSITIVE_INFINITY    // initialize above threshold

    WHILE grad_norm > epsilon_tol:

        n_iter := n_iter + 1

        // STEP A: Compute epistemic confidence at current state
        p_conf := ComputeConfidence(v_n, n_n, dag)

        // STEP B: Apply AEGIS-PROOF-3.1 — verify monotonicity condition
        //         (ensures mode selection is optimal at this confidence level)
        mode   := SelectOptimalMode(p_conf)     // FILTER if p≥0.954, else FLASH
        mono_ok := VerifyErrorDifferentialMonotonicity(dag, p_conf)
        IF NOT mono_ok:
            EMIT WARNING "Monotonicity condition failed at iter " + n_iter +
                         " — continuing under reduced confidence"
        END IF

        // STEP C: Update learning rate αₙ = α₀ / nᵝ
        alpha_n := DiminishingLearningRate(n_iter)

        // STEP D: Gradient step
        //   (vₙ₊₁, nₙ₊₁) = (vₙ, nₙ) − αₙ · ∇J(vₙ, nₙ)
        grad_J := J.Gradient(v_n, n_n, lambda_w, mu_w, dag)
        v_next := v_n - alpha_n * grad_J.v_component
        n_next := n_n - alpha_n * grad_J.n_component

        // STEP E: Add bounded noise ξₙ ~ N(0, σ²I)
        noise      := SampleNoise(dim=2)
        v_next     := v_next + noise[0]
        n_next     := n_next + noise[1]

        // STEP F: Verify DIRAM constraint ε(transition) ≤ 0.6
        diram_ok := VerifyDIRAMCompatibility(v_next, n_next,
                                              J_STAR_APPROX,
                                              lambda_w, mu_w, dag, n_iter)
        IF NOT diram_ok:
            // DIRAM violation: clamp step to respect ε bound
            v_next := LERP(v_n, v_next, t=0.5)    // halve the step
            n_next := LERP(n_n, n_next, t=0.5)
            EMIT INFO "DIRAM step clamped at iter " + n_iter
        END IF

        // STEP G: Commit update
        v_n := v_next
        n_n := n_next

        // STEP H: Compute gradient norm for convergence check
        grad_J_new := J.Gradient(v_n, n_n, lambda_w, mu_w, dag)
        grad_norm  := NORM(grad_J_new)

        // Record trace entry
        trace_record := PrintAndTraceVerification(v_n, n_n, p_conf,
                                                   n_iter, dag, lambda_w, mu_w)
        trace_log.APPEND(trace_record)
        J_history.APPEND(J.Evaluate(v_n, n_n, lambda_w, mu_w, dag))

        EMIT LOG "n=" + n_iter + " | αₙ=" + alpha_n + " | ‖∇J‖=" + grad_norm +
                 " | mode=" + mode + " | DIRAM=" + (IF diram_ok THEN "✓" ELSE "⚠")

    END WHILE   // convergence achieved when ‖∇J‖ ≤ ε_tol

    // ── RETURN CONVERGED CONFIGURATION ─────────────────────
    J_star_final := J.Evaluate(v_n, n_n, lambda_w, mu_w, dag)
    EMIT INFO "=== CONVERGENCE ACHIEVED | iter=" + n_iter +
              " | J*≈" + J_star_final + " ==="
    EMIT INFO "Convergence guarantee: AEGIS-PROOF-3.2"

    RETURN HybridModeResult(
        success    = TRUE,
        v_star     = v_n,
        n_star     = n_n,
        J_star     = J_star_final,
        iterations = n_iter,
        trace_log  = trace_log,
        J_history  = J_history
    )
END PROCEDURE


PROCEDURE ComputeConfidence(v: Verb, n: Noun, dag: EpistemicDAG) -> REAL:
    // Compute epistemic confidence p_conf at (v, n) in the DAG.
    // Higher confidence → smoother posterior distribution → lower entropy.
    H := ComputeNodeEntropy(dag.node_at(v, n))
    H_max := LOG2(dag.n_interpretations)   // maximum entropy for n classes
    p_conf := 1.0 - (H / MAX(H_max, 1e-12))   // normalized: 1 = fully confident
    RETURN CLAMP(p_conf, 0.0, 1.0)
END PROCEDURE

PROCEDURE VerifyNASACompliance(hybrid_result: HybridModeResult) -> BOOL:
    // Check all four NASA-STD-8739.8 properties against a completed run.

    ALL_OK := TRUE

    // (1) Deterministic: result must be converged
    IF NOT hybrid_result.success:
        EMIT ERROR "NASA(1) Deterministic: convergence not achieved"
        ALL_OK := FALSE
    END IF

    // (2) Bounded: learning rates must have been decreasing
    IF hybrid_result.iterations > 0:
        alpha_final := DiminishingLearningRate(hybrid_result.iterations)
        IF alpha_final >= ALPHA_0:
            EMIT ERROR "NASA(2) Bounded: learning rate did not diminish"
            ALL_OK := FALSE
        END IF
    END IF

    // (3) Graceful: no trace entry should show undefined mode selection
    FOR EACH record IN hybrid_result.trace_log:
        IF record.mode NOT IN [FILTER, FLASH]:
            EMIT ERROR "NASA(3) Graceful: undefined mode at iter " + record.iteration
            ALL_OK := FALSE
        END IF
    END FOR

    // (4) Formal: audit trail present
    IF LENGTH(hybrid_result.trace_log) == 0:
        EMIT ERROR "NASA(4) Formal: no audit trail recorded"
        ALL_OK := FALSE
    END IF

    IF ALL_OK:
        EMIT INFO "NASA-STD-8739.8 compliance verified ✓"
    END IF
    RETURN ALL_OK
END PROCEDURE
